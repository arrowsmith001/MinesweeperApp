     private boolean hasWon() {
         for (int i = 0; i < grid.rows; i++) {
             for (int j = 0; j < grid.columns; j++) {
 
                 final Square square = grid.getSquareAtCoordinate(new Coordinate(i, j));
                 if(!square.hasMine && !square.isRevealed)
                 {
                     return false;
                 }
 
             }
         }
         return true;
     }
